// Carregar a biblioteca do Google Charts
google.charts.load('current', {packages:['corechart']});
google.charts.setOnLoadCallback(configurarDatasIniciais);

function configurarDatasIniciais() {
    alterarPeriodo(); // Set default range correctly
    aplicarFiltro();
}

function alterarPeriodo() {
    var tipo = document.getElementById("filtroPeriodo").value;
    var inpIn = document.getElementById("dataInicio");
    var inpFim = document.getElementById("dataFim");
    
    var hoje = new Date();
    
    if (tipo === "custom") {
        inpIn.disabled = false;
        inpFim.disabled = false;
        return;
    }
    
    inpIn.disabled = true;
    inpFim.disabled = true;
    
    var dtFim = new Date(hoje);
    var dtIn = new Date(hoje);
    
    if (tipo === "hoje") {
        // Já estão iguais
    } else if (tipo === "semana") {
        dtIn.setDate(hoje.getDate() - 7);
    } else if (tipo === "mes") {
        dtIn.setDate(1); // Primeiro dia do mês atual
    }
    
    // Formatar YYYY-MM-DD local
    var isoIn = dtIn.getFullYear() + "-" + String(dtIn.getMonth()+1).padStart(2,'0') + "-" + String(dtIn.getDate()).padStart(2,'0');
    var isoFim = dtFim.getFullYear() + "-" + String(dtFim.getMonth()+1).padStart(2,'0') + "-" + String(dtFim.getDate()).padStart(2,'0');
    
    inpIn.value = isoIn;
    inpFim.value = isoFim;
}

function aplicarFiltro() {
    var userLocal = localStorage.getItem("user") || "null";
    
    var fInicio = document.getElementById("dataInicio").value;
    var fFim = document.getElementById("dataFim").value;
    
    var objFiltro = { inicio: fInicio, fim: fFim };

    // Efeito Visual de Loading
    document.getElementById("kpiPeriodo").innerHTML = "...";
    document.getElementById("kpiQtd").innerHTML = "...";
    document.getElementById("kpiLucro").innerHTML = "...";

    google.script.run
        .withSuccessHandler(function(d) {
            document.getElementById("kpiPeriodo").innerHTML = "$ " + d.totalPeriodo.toFixed(2);
            document.getElementById("kpiQtd").innerHTML = d.qtdVendasPeriodo;
            document.getElementById("kpiLucro").innerHTML = "$ " + d.lucroPeriodo.toFixed(2);

            crearGrafico("chartVendedores", d.vendedores.map(v => [v.nome, v.total]), "ARS x Vendedor");
            crearGrafico("chartServicios", d.servicios, "ARS x Serviço");
            crearPie("chartMonedas", d.monedas, "Volume por Moeda (Original)");
            crearGrafico("chartDias", d.ventasPorDia, "ARS x Dia");
        })
        .getDashboardEmpresa(userLocal, objFiltro);
}

function crearGrafico(id, data, title) {
    var dt = new google.visualization.DataTable();
    dt.addColumn('string', 'Item');
    dt.addColumn('number', 'Valor');
    data.forEach(r => dt.addRow(r));

    var chart = new google.visualization.ColumnChart(document.getElementById(id));
    chart.draw(dt, {
        title: title,
        colors: ['#2980b9'],
        legend: { position: "none" },
        chartArea: { width: '80%', height: '70%' },
        animation:{ startup: true, duration: 1000, easing: 'out' }
    });
}

function crearPie(id, data, title) {
    var dt = new google.visualization.DataTable();
    dt.addColumn('string', 'Item');
    dt.addColumn('number', 'Valor');
    data.forEach(r => dt.addRow(r));

    var chart = new google.visualization.PieChart(document.getElementById(id));
    chart.draw(dt, {
        title: title,
        pieHole: 0.4,
        colors: ['#27ae60', '#f39c12', '#8e44ad', '#e74c3c'],
        chartArea: { width: '90%', height: '80%' },
        animation:{ startup: true, duration: 1000, easing: 'out' }
    });
}
